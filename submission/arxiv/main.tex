\begin{abstract}
We investigate Topological Integrity Gravity (TIG) as a constrained quadratic $f(R)$ extension of General Relativity. The model exhibits a structurally controlled transition in horizon formation governed by a dimensionless critical parameter $\beta_c$.

We derive the horizon structure analytically and analyze modifications in photon-sphere geometry, quasi-normal-mode spectra, and echo signatures. Near the critical regime, nonlinear deviations from Schwarzschild behavior emerge, leading to frequency shifts and parametrically enhanced echo delays.

These features provide potential observational signatures in gravitational-wave and black hole imaging data.
\end{abstract}

\section{Introduction}

General Relativity remains the foundation of modern gravitational physics \cite{einstein1915,wald1984}. Among its extensions, $f(R)$ gravity provides a minimal framework for incorporating higher-curvature corrections \cite{sotiriou2010,defelice2010}. Quadratic curvature corrections are particularly important due to their relation to Starobinsky gravity \cite{starobinsky1980}.

In this work, we study a constrained quadratic $f(R)$ model that exhibits a critical transition in horizon formation. Unlike purely phenomenological regular black hole constructions such as Bardeen or Hayward-type models \cite{bardeen1968,hayward2006}, this transition is characterized by an analytic critical parameter controlling the existence of horizons.

\section{Formalism}

We consider the action

\begin{equation}
S = \int d^4x \sqrt{-g} \left[\frac{1}{16\pi}(R - 2\Lambda + \alpha R^2)\right] + S_{\mathrm{matter}}.
\end{equation}

Defining

\begin{equation}
f(R) = R - 2\Lambda + \mu R^2, \quad \mu = 16\pi\alpha,
\end{equation}

the model corresponds to quadratic $f(R)$ gravity.

\subsection{Scalar Degree of Freedom}

\begin{equation}
\phi = f'(R) = 1 + 2\mu R
\end{equation}

\begin{equation}
m_\phi^2 = \frac{1}{3f''(R)} = \frac{1}{6\mu}
\end{equation}

\section{Field Equations}

\begin{equation}
f'(R)R_{\mu\nu} - \frac{1}{2}f(R)g_{\mu\nu}
- \nabla_\mu \nabla_\nu f'(R)
+ g_{\mu\nu}\Box f'(R)
= 8\pi T_{\mu\nu}
\end{equation}

\section{Static Configurations}

\begin{equation}
ds^2 = -F(r)dt^2 + \frac{dr^2}{F(r)} + r^2 d\Omega^2
\end{equation}

\subsection{Effective Interior Profile}

\begin{equation}
m(r) = \frac{Mr^3}{r^3 + r_c^3}
\end{equation}

\begin{equation}
F(r) = 1 - \frac{2m(r)}{r}
\end{equation}

\section{Horizon Structure}

\begin{equation}
F(r_H) = 0
\end{equation}

\begin{equation}
r_H^3 - 2Mr_H^2 + r_c^3 = 0
\end{equation}

Define

\begin{equation}
x = \frac{r_H}{2M}, \quad \beta = \frac{r_c}{2M}
\end{equation}

\begin{equation}
x^3 - x^2 + \beta^3 = 0
\end{equation}

\subsection{Critical Transition}

\begin{equation}
\beta_c = \left(\frac{4}{27}\right)^{1/3}
\end{equation}

\section{Deviation from Schwarzschild}

\begin{equation}
x \approx 1 - \beta^3
\end{equation}

\begin{equation}
r_H \approx 2M(1 - \beta^3)
\end{equation}

\section{Photon Sphere}

\begin{equation}
rF'(r) - 2F(r) = 0
\end{equation}

\begin{equation}
R_{\text{sh}} = \frac{r_{\text{ph}}}{\sqrt{F(r_{\text{ph}})}}
\end{equation}

\section{Perturbative Dynamics}

\begin{equation}
\frac{d^2\Psi}{dr_*^2} + \left[\omega^2 - V_{\text{eff}}(r)\right]\Psi = 0
\end{equation}

\begin{equation}
\frac{dr_*}{dr} = \frac{1}{F(r)}
\end{equation}
