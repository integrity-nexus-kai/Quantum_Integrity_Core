\begin{abstract}
Topological Integrity Gravity (TIG) is a modified gravitational theory based on a quadratic curvature correction to General Relativity. In its minimal realization, the theory corresponds to an $R^2$ extension of the gravitational action, leading to well-defined higher-curvature field equations while preserving classical vacuum solutions. We derive the modified field equations, analyze their implications for black hole configurations, and obtain explicit cosmological predictions. The framework remains consistent with known weak-field limits while introducing dynamically activated corrections in evolving curvature regimes.
\end{abstract}

% --------------------------------------------------
\section{Introduction}

General Relativity provides a geometrical description of gravitation in terms of spacetime curvature \cite{einstein1915, wald1984}. Modified gravity theories introduce higher-order curvature terms to account for high-energy and cosmological regimes.

In this work, we present Topological Integrity Gravity (TIG) in its minimal realization, corresponding to a quadratic curvature extension.

% --------------------------------------------------
\section{Formalism}

\begin{equation}
S = \int d^4x \sqrt{-g}
\left[
\frac{1}{16\pi}(R - 2\Lambda) + \alpha R^2
\right]
+ S_{\mathrm{matter}}
\end{equation}

\begin{equation}
f(R) = R - 2\Lambda + \mu R^2, \quad \mu = 16\pi\alpha
\end{equation}

% --------------------------------------------------
\section{Field Equations}

\begin{equation}
f'(R) R_{\mu\nu}
- \frac{1}{2} f(R) g_{\mu\nu}
+ (g_{\mu\nu}\Box - \nabla_\mu \nabla_\nu) f'(R)
= 8\pi T_{\mu\nu}
\end{equation}

\begin{equation}
f'(R) = 1 + 2\mu R
\end{equation}

% --------------------------------------------------
\section{Limit Analysis}

\textbf{GR limit:}
\begin{equation}
\alpha \to 0 \Rightarrow G_{\mu\nu} = 8\pi T_{\mu\nu}
\end{equation}

\textbf{Vacuum limit:}
\begin{equation}
R = 0 \Rightarrow \text{Schwarzschild solution preserved}
\end{equation}

\textbf{Weak-field limit:}

Linearizing around Minkowski space yields Newtonian gravity with corrections of order $\alpha$.

% --------------------------------------------------
\section{Black Hole Solutions}

In vacuum ($T_{\mu\nu}=0$) and $R=0$, classical Schwarzschild solutions remain valid. Deviations arise only in non-Ricci-flat or dynamical curvature regimes.

\section{Cosmology}

\begin{equation}
ds^2 = -dt^2 + a(t)^2 d\vec{x}^2
\end{equation}

\begin{equation}
R = 6(\dot{H} + 2H^2)
\end{equation}

\begin{equation}
H^2 =
\frac{
8\pi \rho + \Lambda + 8\pi\alpha R^2 - 48\pi\alpha H \dot{R}

3(1 + 32\pi\alpha R)
}
\end{equation}

\subsection{Explicit Cosmological Prediction}

\begin{equation}
\dot{R} = 6(\ddot{H} + 4H\dot{H})
\end{equation}

\begin{equation}
3H^2 =
8\pi\rho + \Lambda
+
288\pi\alpha
\left(
\dot{H}^2 - 2H^2\dot{H} - H\ddot{H}
\right)
\end{equation}

Cosmological deviations arise from time variation of curvature rather than curvature magnitude.

% --------------------------------------------------
\subsection{Numerical Prediction for the Expansion History}

We define

\begin{equation}
E(z)^2 = \frac{H(z)^2}{H_0^2}
\end{equation}

\begin{equation}
E_{\Lambda CDM}^2(z) = \Omega_m (1+z)^3 + \Omega_\Lambda
\end{equation}

\begin{equation}
\Delta E_{\mathrm{TIG}}^2(z)
=
96\pi A
\left[
\frac{3}{4}\Omega_m^2(1+z)^6
-
\frac{3}{2}\Omega_\Lambda \Omega_m (1+z)^3
\right]
\end{equation}

\begin{equation}
A = \alpha H_0^2
\end{equation}

\begin{equation}
H_{\mathrm{TIG}}(z)
=
H_0
\sqrt{
\Omega_m(1+z)^3 + \Omega_\Lambda
+
96\pi A
\left[
\frac{3}{4}\Omega_m^2(1+z)^6
-
\frac{3}{2}\Omega_\Lambda \Omega_m (1+z)^3
\right]
}
\end{equation}

The dominant correction scales as $(1+z)^6$, indicating strong effects in the early universe and suppression at late times.

\section{Stability Analysis}

The minimal $R^2$ model corresponds to a known class of $f(R)$ gravity. For $\alpha > 0$, the additional scalar degree of freedom is non-tachyonic, and classical ghost instabilities are avoided \cite{defelice2010}.

% --------------------------------------------------
\section{Observational Consistency}

Ricci-flat vacuum solutions remain unchanged, ensuring compatibility with gravitational wave observations \cite{gwtc3}.

Cosmological deviations may be constrained by CMB measurements \cite{planck2018}.

% --------------------------------------------------
\section{Conclusion}

We presented TIG as a quadratic curvature extension of General Relativity with explicit field equations and cosmological predictions. The theory preserves known limits while introducing dynamically activated corrections, providing a testable modification of gravitational physics.
